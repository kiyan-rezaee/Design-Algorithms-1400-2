def greedy(n, m, p):
    ans = 0
    machines = [[0] for _ in range(m)]
    for w, i in zip(p, range(n)):
        machines[i % m].append(w + machines[i % m][-1])
        ans += machines[i % m][-1]
    m = [0]
    return ans
